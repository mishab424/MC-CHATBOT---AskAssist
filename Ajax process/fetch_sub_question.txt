DECLARE
  l_json CLOB;
BEGIN
  SELECT JSON_ARRAYAGG(
           JSON_OBJECT(
             'id' VALUE SUB_QUESTION_ID,
             'question' VALUE SUB_QUESTION
           )
         )
  INTO l_json
  FROM ASNT_SMS_CHATBOT_SUB_QUESTION
  WHERE MAIN_QUESTION_ID = TO_NUMBER(APEX_APPLICATION.G_X01);

  SYS.HTP.P(l_json);
END;
